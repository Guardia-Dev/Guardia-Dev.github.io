class Solution:
    def dominantIndex(self, nums):
        if len(nums) is 0:
            return -1
        if len(nums) is 1:
            return 0
        _nums = sorted(nums)
        _max = _nums[-1]
        _max2 = _nums[-2]
        if _max2 * 2 <= _max:
            for i in range(0, len(nums)):
                if _max == nums[i]:
                    return i
        else :
            return -1

from collections import deque
class Solution:
    def openLock(self, deadends, target):
        deadents_filter = set(deadends)
        step, end_flag = 0, 'e'
        visited, queue = set(), deque(['0000', end_flag])

        while queue:
            code = queue.popleft()
            # print(code)
            if code == target:
                return step
            if code in visited or code in deadents_filter:
                continue
            if code == end_flag and not queue:
                return -1
            if code == end_flag:
                queue.append(end_flag)
                step += 1
            else:
                visited.add(code)
                queue.extend(self.nextList(code))
        return -1

    def nextList(self, code):
        res = []
        for ind, let in enumerate(code):
            num = int(let)
            res.append(code[:ind] + str(9 if num == 0 else num - 1) + code[ind + 1:])
            res.append(code[:ind] + str(0 if num == 9 else num + 1) + code[ind + 1:])
        return res

class Solution:
    def maxArea(self, height):
        l, r = 0, len(height) - 1
        res = 0
        while l < r:
            res = max(res, (r - l) * min(height[l], height[r]))
            if height[l] < height[r]:
                l += 1
            else :
                r -= 1
        return res
