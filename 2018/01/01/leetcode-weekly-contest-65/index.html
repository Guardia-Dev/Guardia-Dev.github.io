import math
class Solution:
    def reachNumber(self, target):
        target_abs = abs(target)
        n = math.ceil((-1 + math.sqrt(1 + 8 * target_abs)) / 2)
        sum = n * (n + 1) / 2
        if sum == target:
            return n
        res = int(sum - target)
        if res & 1 == 0:
            return n
        else :
            return n + (1 if n & 1 == 0 else 2)
